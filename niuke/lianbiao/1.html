反转链表

/**
 * struct ListNode {
 *  int val;
 *  struct ListNode *next;
 *  ListNode(int x) : val(x), next(nullptr) {}
 * };
 */

class Solution {
  public:
    ListNode* ReverseList(ListNode* head) {
        // prev为已翻转的最后一个结点
        // cur为当前待翻转结点
        // next保存后续的结点
        ListNode* prev = nullptr, *cur = head, *next = nullptr;
        while (cur != nullptr) {
            next = cur->next;
            cur->next = prev;
            prev = cur;
            cur = next;
        }
        return prev;
    }
};
